# 分布式推理仿真测试计划

**创建日期**: 2026-07-11  
**目标**: 建立完整的端到端仿真测试框架，验证分布式推理在真实环境下的可用性  
**约束**: 必须启动真实后端，执行真实推理，不使用 Mock 数据

---

## 一、概述

### 1.1 背景

当前项目的分布式推理功能缺乏端到端的集成测试。现有的单元测试主要验证各个模块的逻辑正确性，但无法验证：

- 真实后端启动后的完整工作流程
- 主从节点间的实际通信和协调
- 从节点离线时的降级处理能力
- 真实用户输入场景下的系统表现

### 1.2 目标

建立一套**仿真测试框架**，实现：

1. **真实后端启动**：启动完整的 `api_server.py`，包含所有模块
2. **真实推理执行**：调用真实的大模型进行推理，不使用 Mock 响应
3. **人机交互模拟**：模拟真实用户的输入模式（打字速度、停顿、修改等）
4. **分布式验证**：验证主从节点协作推理的正确性
5. **降级能力验证**：在无从节点时验证单机推理的可用性

### 1.3 测试范围

| 测试类型 | 覆盖场景 | 优先级 |
|----------|----------|--------|
| 单机推理测试 | 无从节点时的完整推理流程 | P0 |
| 分布式推理测试 | 主从节点协作推理 | P1 |
| 降级测试 | 从节点离线时的自动降级 | P1 |
| 压力测试 | 多用户并发请求 | P2 |
| 异常测试 | 网络中断、节点崩溃等 | P2 |

#### 2.2.3 人机模拟器（HumanSimulator）

**职责**：
- 模拟真实用户的输入行为
- 控制输入速度（打字速度、停顿）
- 模拟用户修改（删除、重新输入）
- 模拟用户中断（取消生成）

**实现**：
```python
class HumanSimulator:
    def __init__(self, typing_speed: float = 5.0, pause_probability: float = 0.3):
        """
        Args:
            typing_speed: 打字速度（字符/秒）
            pause_probability: 停顿概率（0-1）
        """
        self.typing_speed = typing_speed
        self.pause_probability = pause_probability
    
    async def simulate_input(self, text: str) -> AsyncGenerator[str, None]:
        """
        模拟逐字符输入，包含随机停顿
        
        Yields:
            每次输入的字符或字符串片段
        """
        for i, char in enumerate(text):
            # 模拟打字延迟
            delay = 1.0 / self.typing_speed
            
            # 随机停顿（模拟思考）
            if random.random() < self.pause_probability:
                delay += random.uniform(0.5, 2.0)
            
            await asyncio.sleep(delay)
            yield char
    
    async def simulate_modification(self, original: str, modified: str):
        """模拟用户修改输入（删除 + 重新输入）"""
        # 删除部分字符
        delete_count = len(original) - len(os.path.commonprefix([original, modified]))
        for _ in range(delete_count):
            yield {"action": "delete"}
            await asyncio.sleep(0.1)
        
        # 输入新字符
        new_part = modified[len(os.path.commonprefix([original, modified])):]
        for char in new_part:
            yield {"action": "input", "char": char}
            await asyncio.sleep(1.0 / self.typing_speed)
```

#### 2.2.4 请求发送器（RequestSender）

**职责**：
- 发送 HTTP 请求到后端
- 处理流式响应（SSE）
- 验证响应格式和内容
- 记录性能指标

**实现**：
```python
class RequestSender:
    def __init__(self, base_url: str = "http://localhost:8000"):
        self.base_url = base_url
        self.session = httpx.AsyncClient()
    
    async def send_chat_request(
        self,
        message: str,
        session_id: str = "test-session",
        stream: bool = True,
    ) -> AsyncGenerator[Dict, None]:
        """
        发送聊天请求并处理响应
        
        Args:
            message: 用户消息
            session_id: 会话ID
            stream: 是否使用流式响应
            
        Yields:
            响应片段（流式）或完整响应
        """
        endpoint = f"{self.base_url}/api/chat/stream" if stream else f"{self.base_url}/api/chat"
        
        payload = {
            "message": message,
            "session_id": session_id,
            "stream": stream,
        }
        
        start_time = time.time()
        
        if stream:
            async with self.session.stream("POST", endpoint, json=payload) as response:
                async for line in response.aiter_lines():
                    if line.startswith("data: "):
                        data = json.loads(line[6:])
                        yield {
                            "type": "chunk",
                            "content": data.get("content", ""),
                            "latency": time.time() - start_time,
                        }
        else:
            response = await self.session.post(endpoint, json=payload)
            data = response.json()
            yield {
                "type": "complete",
                "content": data.get("content", ""),
                "latency": time.time() - start_time,
                "tokens": data.get("tokens_generated", 0),
            }
    
    async def check_health(self) -> Dict:
        """检查后端健康状态"""
        response = await self.session.get(f"{self.base_url}/api/health")
        return response.json()
    
    async def get_cluster_status(self) -> Dict:
        """获取集群状态"""
        response = await self.session.get(f"{self.base_url}/api/cluster/status")
        return response.json()
```

```python
async def test_single_node_inference():
    """单机推理完整测试"""
    
    # 1. 启动后端（仅主节点）
```

```python
    """分布式推理测试"""
    
    # 1. 启动后端（主节点 + 从节点）

    # 3. 获取分层配置
    layer_config = await sender.get_layer_config()
    print(f"分层配置: {layer_config}")
    assert len(layer_config["assignments"]) == 4, "应该有4个节点参与推理"
    
    # 4. 执行推理测试
    simulator = HumanSimulator(typing_speed=5.0)
    scenario = SCENARIOS["complex_reasoning"]

    for question in scenario["questions"]:

        # 模拟用户输入
        full_input = ""
        async for char in simulator.simulate_input(question):
            full_input += char
        
        # 发送请求并监控分布式推理过程
        response_text = ""
        start_time = time.time()
        
        # 启动后台监控（检查各节点负载）
        monitor_task = asyncio.create_task(monitor_node_metrics(sender))

            if chunk["type"] == "chunk":
                response_text += chunk["content"]
                print(chunk["content"], end="", flush=True)

        # 停止监控
        metrics = await monitor_task
        
        # 验证分布式推理
        assert len(response_text) > 0, "响应不应为空"
        assert metrics["nodes_involved"] == 4, "应该有4个节点参与推理"
        assert metrics["layer_forward_count"] > 0, "应该执行了层前向传播"
        
        results.append({
            "question": question,
            "response": response_text,
            "latency": latency,
            "metrics": metrics,
        })
        
        print(f"\n延迟: {latency:.2f}s, 参与节点: {metrics['nodes_involved']}")
    
    # 5. 生成测试报告
    report = TestReport(results)
    report.save("test_results/distributed_inference.json")
    
    # 6. 清理
    await orchestrator.teardown()

async def monitor_node_metrics(sender: RequestSender) -> Dict:
    """监控推理过程中的的节点指标"""
    metrics = {
        "nodes_involved": 0,
        "layer_forward_count": 0,
        "network_traffic": 0,
    }
    
    # 每秒查询一次集群状态
    for _ in range(10):  # 最多监控10秒
        status = await sender.get_cluster_status()
        
        # 统计忙碌节点
        busy_nodes = sum(1 for s in status["slaves"] if s["state"] == "busy")
        metrics["nodes_involved"] = max(metrics["nodes_involved"], busy_nodes + 1)  # +1 主节点
        
        await asyncio.sleep(1)
    
    return metrics
```

**测试流程**：

```python
async def test_concurrent_requests():
    """并发请求压力测试"""
    
    # 1. 启动后端
    orchestrator = TestOrchestrator()
    await orchestrator.setup(TestConfig(
        start_master=True,
        start_slaves=True,
        slave_count=2,
    ))
    
    sender = RequestSender()
    
    # 2. 定义并发场景
    concurrent_users = 5
    requests_per_user = 3
    
    # 3. 生成用户请求
    user_requests = []
    for user_id in range(concurrent_users):
        session_id = f"test-session-{user_id}"
        for req_id in range(requests_per_user):
            question = f"用户{user_id}的问题{req_id}: 什么是机器学习？"
            user_requests.append({
                "session_id": session_id,
                "question": question,
            })
    
    # 4. 并发执行请求
    print(f"\n开始并发测试: {concurrent_users} 用户, 每用户 {requests_per_user} 请求")
    
    tasks = []
    for req in user_requests:
        task = asyncio.create_task(
            execute_inference(sender, req["question"], req["session_id"])
        )
        tasks.append(task)
    
    # 等待所有请求完成
    results = await asyncio.gather(*tasks, return_exceptions=True)
    
    # 5. 分析结果
    success_count = sum(1 for r in results if not isinstance(r, Exception))
    failure_count = len(results) - success_count
    
    latencies = [r["latency"] for r in results if not isinstance(r, Exception)]
    avg_latency = sum(latencies) / len(latencies) if latencies else 0
    max_latency = max(latencies) if latencies else 0
    
    print(f"\n测试结果:")
    print(f"  成功: {success_count}/{len(results)}")
    print(f"  失败: {failure_count}/{len(results)}")
    print(f"  平均延迟: {avg_latency:.2f}s")
    print(f"  最大延迟: {max_latency:.2f}s")
    
    # 6. 验证
    assert success_count / len(results) >= 0.95, "成功率应该 >= 95%"
    assert avg_latency < 15, "平均延迟应该 < 15s"
    
    # 7. 生成报告
    report = StressTestReport({
        "concurrent_users": concurrent_users,
        "total_requests": len(results),
        "success_count": success_count,
        "failure_count": failure_count,
        "avg_latency": avg_latency,
        "max_latency": max_latency,
    })
    report.save("test_results/stress_test.json")
    
    # 8. 清理
    await orchestrator.teardown()
```

**验证指标**：

| 指标 | 预期值 | 说明 |
|------|--------|------|
| 成功率 | ≥ 95% | 允许少量失败 |
| 平均延迟 | < 15s | 并发下可接受 |
| 最大延迟 | < 30s | 最坏情况 |
| 系统稳定性 | 无崩溃 | 后端不崩溃 |

```python
class BackendManager:
    """后端服务管理器"""
    
    def __init__(self):
        self.master_process: Optional[subprocess.Popen] = None
        self.slave_processes: List[subprocess.Popen] = []
    
    async def start_master(
        self,
        config_path: str = ".env",
        port: int = 8000,
    ) -> None:
        """启动主节点后端"""
        
        # 设置环境变量
        env = os.environ.copy()
        env["QLH_NODE_ROLE"] = "master"
        env["QLH_SERVER_PORT"] = str(port)
        
        # 启动进程
        self.master_process = subprocess.Popen(
            ["python", "src/api_server.py"],
            env=env,
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE,
        )
        
        # 等待启动完成
        await self._wait_for_startup(port, timeout=30)
    
    async def start_slave(
        self,
        master_host: str = "localhost",
        master_port: int = 8888,
        slave_port: int = 8001,
    ) -> None:
        """启动从节点后端"""
        
        env = os.environ.copy()
        env["QLH_NODE_ROLE"] = "slave"
        env["QLH_MASTER_HOST"] = master_host
        env["QLH_MASTER_PORT"] = str(master_port)
        env["QLH_SERVER_PORT"] = str(slave_port)
        
        process = subprocess.Popen(
            ["python", "src/api_server.py"],
            env=env,
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE,
        )
        
        self.slave_processes.append(process)
        
        # 等待从节点注册
        await self._wait_for_registration(slave_port, timeout=30)
    
    async def _wait_for_startup(self, port: int, timeout: int) -> None:
        """等待后端启动完成"""
        start_time = time.time()
        
        while time.time() - start_time < timeout:
            try:
                async with httpx.AsyncClient() as client:
                    response = await client.get(f"http://localhost:{port}/api/health")
                    if response.status_code == 200:
                        return
            except httpx.ConnectError:
                pass
            
            await asyncio.sleep(1)
        
        raise TimeoutError(f"后端启动超时（{timeout}秒）")
    
    async def _wait_for_registration(self, port: int, timeout: int) -> None:
        """等待从节点注册完成"""
        start_time = time.time()
        
        while time.time() - start_time < timeout:
            try:
                async with httpx.AsyncClient() as client:
                    response = await client.get(f"http://localhost:{port}/api/cluster/status")
                    if response.status_code == 200:
                        status = response.json()
                        if status.get("mode") == "distributed":
                            return
            except httpx.ConnectError:
                pass
            
            await asyncio.sleep(1)
        
        raise TimeoutError(f"从节点注册超时（{timeout}秒）")
    
    async def stop_all(self) -> None:
        """停止所有后端进程"""
        if self.master_process:
            self.master_process.terminate()
            self.master_process.wait(timeout=5)
        
        for process in self.slave_processes:
            process.terminate()
            process.wait(timeout=5)
```

```python
class ResponseValidator:
    """响应验证器"""
    
    @staticmethod
    def validate_response(
        response: str,
        question: str,
        expected_length: Tuple[int, int] = (50, 500),
    ) -> ValidationResult:
        """验证响应质量"""
        
        errors = []
        warnings = []
        
        # 1. 长度验证
        if len(response) < expected_length[0]:
            errors.append(f"响应过短: {len(response)} < {expected_length[0]}")
        elif len(response) > expected_length[1]:
            warnings.append(f"响应过长: {len(response)} > {expected_length[1]}")
        
        # 2. 相关性验证（简单关键词匹配）
        question_keywords = extract_keywords(question)
        response_lower = response.lower()
        
        keyword_match_count = sum(
            1 for kw in question_keywords if kw in response_lower
        )
        
        if keyword_match_count == 0:
            warnings.append("响应与问题关键词无匹配")
        
        # 3. 连贯性验证（检查是否有明显错误）
        if "error" in response_lower or "错误" in response:
            warnings.append("响应包含错误信息")
        
        if "无法回答" in response or "不知道" in response:
            warnings.append("模型表示无法回答")
        
        # 4. 格式验证
        if response.strip() == "":
            errors.append("响应为空")
        
        return ValidationResult(
            is_valid=len(errors) == 0,
            errors=errors,
            warnings=warnings,
        )
    
    @staticmethod
    def validate_distributed_inference(
        metrics: Dict,
        expected_nodes: int,
    ) -> ValidationResult:
        """验证分布式推理执行"""
        
        errors = []
        warnings = []
        
        # 1. 节点参与数
        if metrics["nodes_involved"] != expected_nodes:
            errors.append(
                f"参与节点数不符: {metrics['nodes_involved']} != {expected_nodes}"
            )
        
        # 2. 层前向传播
        if metrics["layer_forward_count"] == 0:
            errors.append("未执行层前向传播")
        
        # 3. 网络流量
        if metrics["network_traffic"] == 0:
            warnings.append("未检测到网络流量")
        
        return ValidationResult(
            is_valid=len(errors) == 0,
            errors=errors,
            warnings=warnings,
        )
```
